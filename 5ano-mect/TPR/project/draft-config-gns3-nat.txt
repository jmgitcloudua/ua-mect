---------[ Configuration Steps ]---------

- Connect Router with GNS3 NAT cloud

 R1(config)# interface f2/0         : interface connected to NAT clould
 R1(config-if)# ip address dhcp
 R1(config-if)# no shutdown

 R1(config)# interface f0/0         : interface connected to LAN 1
 R1(config-if)# ip address 192.168.1.1 255.255.255.0
 R1(config-if)# no shutdown

 R1# show ip interface brief        : check interface status



 R1(config)# do show ip route    : Shows LAN network + NAT network too + add default route

 R1(config)# interface f0/0         : interface connected to LAN 1
 R1(config-if)# ip nat inside

 R1(config)# interface f2/0         : interface connected to NAT clould
 R1(config-if)# ip nat outside

 R1(config)# ip nat inside source list NAT pool clould overload

 R1(config)# ip nat pool clould 192.168.122.1 192.168.122.255 netmask 255.255.255.0
 R1(config)# ip access-list extended NAT
 R1(config-ext-acl)# permit ip 192.168.1.0 0.0.0.255 any

 R1# show ip nat translations        : See the NAT ing translations


[root@PC-1 /]# vi /etc/network/interfaces

 # Static config for eth0
 auto eth0
 iface eth0 inet static
    address 192.168.1.10
    netmask 255.255.255.0
    gateway 192.168.1.1
    up echo nameserver 192.168.122.1  &gt;  /etc/resolv.conf
